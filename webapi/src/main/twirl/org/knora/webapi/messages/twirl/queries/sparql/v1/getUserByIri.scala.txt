@import org.knora.webapi.IRI

@**
 * Gets all information about a user, given the user's IRI.
 *
 * @param userIri the user's IRI.
 *@
@(userIri: IRI)

SELECT ?s ?p ?o

WHERE {
    BIND(IRI("@userIri") as ?s)
    ?s ?p ?o .
}
