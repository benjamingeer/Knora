@import java.time.Instant
@import org.knora.webapi.IRI

@**
 * Gets the the version history of the values of a resource, using GraphDB.
 *
 * @param resourceIri    the IRI of the resource.
 * @param maybeStartDate the start of the time period to return, inclusive.
 * @param maybeEndDate   the end of the time period to return, exclusive.
 *@
@(resourceIri: IRI,
  maybeStartDate: Option[Instant] = None,
  maybeEndDate: Option[Instant] = None)

PREFIX knora-base: <http://www.knora.org/ontology/knora-base#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT DISTINCT ?timestamp ?user WHERE {
    <@resourceIri> knora-base:hasValue ?currentValue .
    ?currentValue knora-base:previousValue* ?valueObject .

    {
        ?valueObject knora-base:valueCreationDate ?timestamp .
        ?valueObject knora-base:attachedToUser ?user .
    } UNION {
        ?valueObject knora-base:deleteDate ?timestamp .
        ?valueObject knora-base:deletedBy ?user .
    }

    @maybeStartDate match {
        case Some(startDate) => {
            FILTER(?timestamp >= "@startDate"^^xsd:dateTime)
        }

        case None => {}
    }

    @maybeEndDate match {
        case Some(endDate) => {
            FILTER(?timestamp <  "@endDate"^^xsd:dateTime)
        }

        case None => {}
    }
} ORDER BY DESC(?timestamp)
