@import org.knora.webapi._
@import org.knora.webapi.util.SmartIri

@*
 * Constructs a Gravsearch query that gets resources from the specified project.
 *
 * @param projectIri the IRI of the project.
 * @param propertyIris the IRIs of the properties to be queried.
 * @param orderByProperty the IRI of the property to order by.
 * @param page the page number to be returned.
 *@
@(projectIri: IRI,
  resourceClassIri: SmartIri,
  propertyIris: Seq[SmartIri],
  orderByProperty: SmartIri,
  page: Int)

PREFIX knora-api: <http://api.knora.org/ontology/knora-api/v2#>

CONSTRUCT {
  ?resource knora-api:isMainResource true .

  @for((propertyIri, index) <- propertyIris.zipWithIndex) {
    ?resource <@propertyIri> ?propertyObj@index .
  }
} WHERE {
  ?resource a <@resourceClassIri> ;
    knora-api:attachedToProject <@projectIri> .

  @for((propertyIri, index) <- propertyIris.zipWithIndex) {
    OPTIONAL {
      ?resource <@propertyIri> ?propertyObj@index .
    }
  }
}
ORDER BY ?propertyObj@{propertyIris.zipWithIndex.find(_._1 == orderByProperty).get._2}
OFFSET @page
